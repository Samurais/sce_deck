58  	 * @param instType
59  	 * @param swRepoHost
60  	 * @param parms
61  	 * @throws ImageBuilderException
62  	 */
63  	public AbstractWindowsImageBuilder(final String bpmImageId,
64  			final String bpmImageLoc, final String bpmInstanceName,
65  			final String instType, final String swRepoHost,
66  			final Map<String, String> parms) throws ImageBuilderException {
67  		// check parms
68  		if ((!parms.containsKey("UserName"))
69  				|| (!parms.containsKey("Password"))) {
70  			throw new ImageBuilderException(
71  					ImageBuilderException.PARAMS_NOT_VALID);
72  		} else {
73  			if (((String) parms.get("UserName")).isEmpty()
74  					|| ((String) parms.get("Password")).isEmpty()) {
75  				throw new ImageBuilderException(
76  						ImageBuilderException.PARAMS_NOT_VALID);
77  			} else {
78  				setInstUser((String) parms.get("UserName"));
79  				setInstPassword((String) parms.get("Password"));
80  			}
81  		}
82  		setBpmImageId(bpmImageId);
83  		setBpmImageLoc(bpmImageLoc);
84  		setBpmInstanceName(bpmInstanceName);
85  		setInstType(instType);
86  		setSwRepoHost(swRepoHost);
87  		setParms(parms);
88  		// mgrInst = new ManagedInstance(getBpmInstanceName(), getInstType(),
89  		// getBpmImageId(), getBpmImageLoc(), "SCEDeck", parms);
90  		mgrInst = new ManagedInstance(getBpmInstanceName(), getInstType(),
91  				getBpmImageId(), getBpmImageLoc(), null, parms);

334 	 * <p>
335 	 * Title: uploadInstallationScripts
336 	 * </p>
337 	 * <p>
338 	 * Description:
339 	 * </p>
340 	 * 
341 	 * @throws FileNotFoundException
342 	 * @throws ManagedInstanceException
343 	 */
344 	public void uploadInstallationScripts() throws FileNotFoundException,
345 			ManagedInstanceException {
346 		/*
347 		 * make sure the target file path exist in Instance
348 		 */
349 		if (!getMgrInst().isPathExist(installer2InstancePath)) {
350 			try {
351 				getMgrInst().exec("mkdir -p " + installer2InstancePath);
352 			} catch (ManagedInstanceException e) {
353 				mgrLogger.err(className, e.getMessage());
354 			}
355 		}
356 		/**
357 		 * upload the unzip tool
358 		 */
359 		if (!getMgrInst().isPathExist(installer2InstancePath + "/UNZIP.EXE")) {
360 			uploadLocalFile2BaseInstance(installer2InstancePath,
361 					"bpm_windows_image/tools/UNZIP.EXE");
362 			try {
363 				getMgrInst()
364 						.exec("mv " + "UNZIP.EXE " + installer2InstancePath);
365 			} catch (ManagedInstanceException e) {
366 				mgrLogger.err(className, e.getMessage());
367 			}
368 		}
369 		/*
370 		 * upload installation scripts into instance
371 		 */
372 		String installationScripts = mgrProps
373 				.getProperty("bpm_installation_scripts");
374 		// clean old zip file
375 		File f = new File(installationScripts);
376 		String installationScriptsZip = f.getName() + ".zip";
377 		File zipf = new File(installationScriptsZip);
378 		if (zipf.exists()) {
379 			zipf.delete();
380 		}
381 		// package zip -zipf
382 		ZipHandler zip = new ZipHandler();
383 		zip.doZip(installationScripts);
384 
385 		if (zipf.exists()) {
386 			mgrLogger.info(className, "upload bpm installation scripts");
387 			uploadLocalFile2BaseInstance(installer2InstancePath,
388 					installationScriptsZip);
389 			zipf.delete();
390 			mgrLogger.info(className,
391 					"finish uploading bpm installation scripts");
392 			// TODO use parameters , currently the script file is hard code
393 			mgrLogger
394 					.debug(className,
395 							"extract the installation scripts to C:\\installer\\Script .");
396 			getMgrInst().exec(
397 					"cd " + installer2InstancePath + "&&"
398 							+ installer2InstancePath + "/unzip.exe "
399 							+ installationScriptsZip);
400 
401 			getMgrInst().exec(
402 					"mv " + installer2InstancePath
403 							+ "/bpm_windows_image/installer/ "
404 							+ installer2InstancePath + "/Script");
405 			getMgrInst().exec(
406 					"cd " + installer2InstancePath
407 							+ "/Script&&chmod -R 777 `pwd`");
408 			mgrLogger.debug(className,
409 					"extracting the installation scripts is done.");
410 			mgrLogger.debug(className,
411 					"copy kms_license and autolog.bat to C:\\bat .");
412 			getMgrInst().exec(
413 					"cp " + installer2InstancePath
414 							+ "/Script/activation_kms.bat /cygdrive/c/bat");
415 			getMgrInst().exec(
416 					"cp " + installer2InstancePath
417 							+ "/Script/autolog.bat /cygdrive/c/bat");
418 			mgrLogger
419 					.debug(className, "copy remove_cygwin.bat to C:\\cygwin .");
420 			getMgrInst().exec(
421 					"cp " + installer2InstancePath
422 							+ "/Script/remove_cygwin.bat /cygdrive/c/cygwin");
423 
424 		} else {
425 			throw new FileNotFoundException(installationScriptsZip);
426 		}
427 
428 	}

487 	 * Title: beforeSavingWindowsInst2Image

489 	 * <p>
490 	 * Description: do some tasks before saving windows instance, or the image
491 	 * would not work well.
492 	 * </p>
493 	 */
494 	protected boolean beforeSavingWindowsInst2Image() {
495 		boolean succ = false;
496 		try {
497 			/*
498 			 * sync time
499 			 */
500 			mgrLogger.info(className, "Sync Time");
501 			mgrLogger.info(className, getMgrInst()
502 					.exec("w32tm /resync /nowait").getOut());
503 			Thread.sleep(1000 * 60);
504 			/*
505 			 * call slmgr sliently in windows instance
506 			 */
507 			mgrLogger.info(className, "run slmgr to activate ms license");
508 			getMgrInst().exec("/cygdrive/c/bat/activation_kms.bat");
509 			Thread.sleep(1000 * 60);
510 			/*
511 			 * run autolog
512 			 */
513 			mgrLogger.info(className, "run AUTOLOG.BAT");
514 			getMgrInst().exec(
515 					"nohup /cygdrive/c/bat/autolog.bat " + getInstPassword()
516 							+ " &");
517 			boolean isDone = false;
518 			while (!isDone) {
519 				Thread.sleep(60000 * 2);
520 				if (getMgrInst().isPathExist("/cygdrive/c/cygwin/autolog.done")) {
521 					isDone = true;
522 				}
523 				if (getMgrInst().isPathExist("/cygdrive/c/cygwin/autolog.fail")) {
524 					mgrLogger
525 							.err(
526 									className,
527 									"can not execute autolog in windows instance , maybe administrator password is wrong.");
528 					throw new ManagedInstanceException(
529 							ManagedInstanceException.FAIL_TO_RUN_AUTOLOG);
530 				}
531 			}
532 			if (isDone == true) {
533 				mgrLogger.debug(className, "autolog finishes successfully.");
534 				getMgrInst().exec("rm -rf /cygdrive/c/bat/autolog.bat");
535 			}
536 			succ = isDone;
537 		} catch (ManagedInstanceException e) {
538 			succ = false;
539 			e.printStackTrace();
540 		} catch (InterruptedException e) {
541 			succ = false;
542 			e.printStackTrace();
543 		}
544 		return succ;
545 	}
