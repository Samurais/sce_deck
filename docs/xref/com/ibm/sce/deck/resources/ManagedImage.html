302 	private String handleExceptions(Exception e) {
303 		String exceptionType = new String();
304 		if (e instanceof UnauthorizedUserException) {
305 			exceptionType = ManagedImageException.UNAUTHORIZEDUSER;
306 		} else if (e instanceof UnknownImageException) {
307 			exceptionType = ManagedImageException.UNKNOWNIMAGE;
308 		} else if (e instanceof SCEClientException) {
309 			exceptionType = ManagedImageException.SCECLIENT;
310 		} else {
311 			exceptionType = ManagedImageException.UNKNOWNERROR;
312 		}
